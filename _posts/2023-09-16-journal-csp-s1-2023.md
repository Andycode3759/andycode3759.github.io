### CCF 快速幂

```c++
#include <cstdio>
using namespace std;
const int MOD = 1e9 + 7;
// 加取模防止溢出

long long quick_power(long long x, long long n)
{
    if (n == 0)
        return 1;
    if (n == 1)
        return x;
    return quick_power(x, n / 2) * quick_power(x, n / 2) % MOD * ((n & 1) == 1 ? x : 1) % MOD;
}

int main()
{
    printf("%lld\n", quick_power(2, 1000000000));
    // 运行时间：4.3s
    return 0;
}
```
